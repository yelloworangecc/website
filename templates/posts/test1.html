2. Samples

import RTS38 as RTS
RTS.SetConfigurationDatabase('d:\\oe\\DataManager\\Databases\\')
RTS.GetVirtualServers()

# home bonder1
RTS.HomeAllBonderAxes(0)

# clamp related
RTS.GetBondPlatePreset()
RTS.HomeClamp(0)
RTS.GetClampStatus(0)
RTS.CloseClamp(0)
RTS.GetClampStatus(0)

# clear assembly
RTS.ClearAssemblyOffsets(0)

# clear device and download one
RTS.ClearDeviceOffsets(0)
xyzt={'x':0,'y':0,'z':0,'t':0}
RTS.AddDeviceOffset(0,0,xyzt,0,0)

# load process program
RTS.LoadParamSet(0,'D:\\oe\\ProcessPrograms\\hc_test.set\\BondHead1.pp')

# download one wire (notice that the devideId and wireId is in view of process program)
RTS.StartWiresDownload(0)
RTS.StartCommandBlock(0)
RTS.AddWire(0,0,0)
RTS.TerminateCommandBlock(0)

# bond wire (notice that the deviceId and wireId is in view of RTS)
RTS.BondWire(0,1,1)
